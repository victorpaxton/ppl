
        // Recursive function to search x in arr[]
elmntSrch: function integer (arr: array [4] of integer, size: integer, x: integer) {
    rec: integer;
  
    size = size - 1;
  
    if (size >= 0) {
        if (arr[size] == x)
            return size;
        else
            rec = elmntSrch(arr, size, x);
    }
    else
        return -1;
  
    return rec;
}
